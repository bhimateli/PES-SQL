📘 SQL Queries

1. Employee Joining Information
Write a query that returns "Joined on" for all employees.
SELECT CONCAT(first_name, " Joined on ", hire_date) 
FROM employees;
2. Name Formatting
Print first letter in lowercase and remaining in uppercase.
SELECT CONCAT(LOWER(SUBSTRING(first_name,1,1)),
UPPER(SUBSTRING(first_name,2)))
FROM employees;
3. Employees Working More Than 30 Years
Display employee name and hire date for employees with >30 years.
SELECT CONCAT(first_name, " ", last_name), hire_date
FROM employees
WHERE DATEDIFF(CURDATE(), hire_date) > 30*365;
4. Salary Grade Classification
Display grade based on salary ranges.
SELECT first_name, salary,
CASE 
    WHEN salary < 3000 THEN 'Grade V'
    WHEN salary BETWEEN 3000 AND 5000 THEN 'Grade IV'
    WHEN salary > 5000 AND salary <= 10000 THEN 'Grade III'
    WHEN salary > 10000 AND salary <= 50000 THEN 'Grade II'
    WHEN salary > 50000 THEN 'Grade I'
END AS Grade
FROM employees;
5. Employees with Same Joining Date
Display employees having same hire date.
SELECT first_name, job_id, hire_date
FROM employees
WHERE hire_date IN (
    SELECT hire_date FROM employees
    GROUP BY hire_date HAVING COUNT(*) > 1
);
6. Maximum Salary per Department
Find maximum salary in each department.
SELECT department_id, MAX(salary)
FROM employees
GROUP BY department_id;
7. Department Headcount
Display employee count per department with department name.
SELECT d.department_name, COUNT(e.employee_id)
FROM employees e
JOIN departments d
ON e.department_id = d.department_id
GROUP BY d.department_name;
8. Set Operator Example
Display employees from department 60 and 90.
SELECT employee_id, first_name, salary, department_id 
FROM employees WHERE department_id = 60
UNION
SELECT employee_id, first_name, salary, department_id 
FROM employees WHERE department_id = 90;
9. Second Highest Salary
Find second highest salary.
SELECT MAX(salary)
FROM employees
WHERE salary NOT IN (SELECT MAX(salary) FROM employees);
10. Above Average Salary
Find employees earning more than department average.
SELECT employee_id, first_name, last_name, salary, department_id
FROM employees e
WHERE salary > (
    SELECT AVG(salary)
    FROM employees
    WHERE department_id = e.department_id
);

🛠️ DDL Statements

11. Create Course Table
Create table with constraints.
CREATE TABLE Course (
    Course_ID INT,
    Course_Name VARCHAR(10),
    course_duration INT,
    course_title VARCHAR(20),
    course_start_date DATE,
    CHECK (course_duration <= 21),
    CHECK (course_title IN ('Python','SQL','STATS'))
);
12. Create View
Create view emp_dept using employees and departments.
CREATE VIEW emp_dept AS
SELECT e.employee_id, e.department_id, d.department_name,
       e.salary, e.commission_pct, e.hire_date
FROM employees e
JOIN departments d
ON e.department_id = d.department_id;
